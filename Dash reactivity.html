## Basic callback -------------------------- 

```
import dash
import dash_core_components as dcc
import dash_html_components as html
from dash.dependencies import Input, Output

app = dash.Dash()

app.layout = html.Div([
    dcc.Input(id='text_in', value='Name Surname', type='text'),
    dcc.Dropdown(
    	id = 'option_in',
        options=[
            {'label': 'Facebook', 'value': 'www.fb.com'},
            {'label': 'Twitter', 'value': 'www.twitter.com'},
            {'label': 'AUA', 'value': 'www.aua.am'}
        ],
        value='www.fb.com'),
    html.Div(id='text_out')
])


@app.callback(
    Output(component_id='text_out', component_property='children'),
    [Input(component_id='text_in', component_property='value'),
     Input(component_id='option_in', component_property='value')]
)
def update_text_output(input_value_1,input_value_2):
    return input_value_1 + ", please follow me on " + input_value_2


if __name__ == '__main__':
    app.run_server()
```
## Callback with State -------------------------- 

```
import dash
import dash_core_components as dcc
import dash_html_components as html
from dash.dependencies import Input, Output, State

app = dash.Dash()

app.layout = html.Div([
    dcc.Input(id='text_in', value='Name Surname', type='text'),
    dcc.Dropdown(
    	id = 'option_in',
        options=[
            {'label': 'Facebook', 'value': 'www.fb.com'},
            {'label': 'Twitter', 'value': 'www.twitter.com'},
            {'label': 'AUA', 'value': 'www.aua.am'}
        ],
        value='www.fb.com'),
    html.Button(id='submit',n_clicks=0, children='Submit')
    html.Div(id='text_out')
])


@app.callback(
    Output(component_id='text_out', component_property='children'),
    [Input(component_id='submit', component_property='n_clicks')],
    [State(component_id='text_in', component_property='value'),
     State(component_id='option_in', component_property='value')]
)
def update_text_output(clicks,input_value_1,input_value_2):
    return input_value_1 + ", please follow me on " + input_value_2


if __name__ == '__main__':
    app.run_server()
```
## Graph with callback -------------------------- 

```
import dash
import dash_core_components as dcc
import dash_html_components as html
from dash.dependencies import Input, Output

import plotly.graph_objs as go
import quandl

app = dash.Dash()

app.layout = html.Div([
    dcc.Dropdown(
    	id = 'option_in',
        options=[
            {'label': 'Google', 'value': 'GOOGL'},
            {'label': 'Apple', 'value': 'AAPL'},
            {'label': 'Microsoft', 'value': 'MSFT'}
        ],
        value='GOOGL'),
    dcc.Graph(id='figure_1')
])


@app.callback(
    Output(component_id='figure_1', component_property='figure'),
    [Input(component_id='option_in', component_property='value')]
)
def update_graph(input_value):
	quandl_input = "WIKI/"+input_value
	stock_data = quandl.get(quandl_input, authtoken = "mytoken")
	x_values = stock_data.index
	y_values = stock_data.Open
	trace = go.Scatter(x=x_values,y=y_values,mode='lines')
	data = [trace]
	figure = dict(data=data)
    return figure


if __name__ == '__main__':
    app.run_server()
```
## Databases
## Comparison ----------------------- ||Database|Spreadsheet| |-|-|-| |**Objective**|Storing|Analysis| |**Tables**|Multiple|Single| |**Data**|Structured|Unstructured| |**Access**|Multiple|Single| |**Manipulation**|External|Internal|
## SQL ----------------------- - stands for Structured Querly Language - is the most widely used database language - is considered standard - is designed for managing data in relational databases - has many variations with tiny differences - has extremely easy syntax Example query: 

```
SELECT column_name FROM table_name
```
### Alternatives to Relational Databases ----------------------- - NoSQL (not only SQL) databases - mostly based on JSON (JavaScript Object Notation) documents - represent key-value pairs - similar to Python dictionaries Example NoSQL database document: 

```
{
    "FirstName": "Hrant", 
    "LastName": "Davtyan", 
    "University": "AUA",
    "Course": "Business Intelligence"
}
```
